    JavaScript (show/hide based on scroll position in courses section):
    ```javascript
    // Sticky CTA visibility
    function updateStickyCta() {
        const coursesSection = document.getElementById('courses');
        const stickyCta = document.getElementById('pricingStickyCta');
        if (!coursesSection || !stickyCta) return;

        const rect = coursesSection.getBoundingClientRect();
        const isInSection = rect.top < window.innerHeight * 0.5 && rect.bottom > 100;

        stickyCta.classList.toggle('visible', isInSection);
    }

    // Throttled scroll listener
    let stickyTicking = false;
    window.addEventListener('scroll', function() {
        if (!stickyTicking) {
            window.requestAnimationFrame(function() {
                updateStickyCta();
                stickyTicking = false;
            });
            stickyTicking = true;
        }
    });

    // Initial check
    updateStickyCta();
    ```
